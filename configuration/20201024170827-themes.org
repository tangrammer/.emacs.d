;; switching themes with keys asigned
(defun theme-clojure ()
  (interactive)
  (load-theme 'solarized-dark t))

(defun theme-org ()
  (interactive)
  ;; theme recommended by toxi for LP
  (load-theme 'leuven t))

(theme-org)
